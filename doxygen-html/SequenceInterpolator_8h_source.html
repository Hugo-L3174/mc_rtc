134  Value compute(double currTime)
135  {
136  if(values_.empty())
137  {
138  mc_rtc::log::error_and_throw<std::runtime_error>("SequenceInterpolator requires at least one value");
139  }
140 
141  // Check for out-of-bound access
142  if(currTime >= values_.back().first)
143  {
144  return values_.back().second;
145  }
146  else if(currTime <= values_.front().first)
147  {
148  return values_.front().second;
149  }
150 
151  /*
152  * Efficiently update interval index and associated cached values:
153  * 1/ Check whether currTime is in the current interval
154  * 2/ Check whether currTime is in the next interval
155  * 3/ Perform a binary search
156  * Note that we don't need to check indices bounds here as they are valid by
157  * construction.
158  */
159  auto updateIndex = [this, currTime]() {
160  if(values_[prevIndex_].first < currTime && values_[nextIndex_].first >= currTime)
161  {
162  return;
163  }
164  else if(values_[nextIndex_].first < currTime && values_[nextIndex_ + 1].first >= currTime)
165  {
166  ++prevIndex_;
167  ++nextIndex_;
168  intervalDuration_ = values_[nextIndex_].first - values_[prevIndex_].first;
169  }
170  else
171  {
172  // first element in values_ with time greater or equal to currTime
173  auto nextIt = std::lower_bound(std::begin(values_), std::end(values_), currTime,
174  [](const TimedValue & lhs, const double & rhs) { return lhs.first < rhs; });
175  nextIndex_ = static_cast<size_t>(nextIt - values_.begin());
176  prevIndex_ = nextIndex_ - 1;
177  intervalDuration_ = values_[nextIndex_].first - values_[prevIndex_].first;
178  }
179  };
180 
181  updateIndex();
182  const auto & prevTime = values_[prevIndex_].first;
183  return interpolator_(values_[prevIndex_].second, values_[nextIndex_].second,
184  (currTime - prevTime) / intervalDuration_);
185  }
