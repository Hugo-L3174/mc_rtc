222  void load(const mc_rtc::Configuration & config)
223  {
224  config("safety_tresholds", safetyThresholds);
225 
226  config("fdqp_weights", fdqpWeights);
227  config("leftFootSurface", leftFootSurface);
228  config("rightFootSurface", rightFootSurface);
229  config("torsoBodyName", torsoBodyName);
230  config("friction", friction);
231 
232  if(config.has("admittance"))
233  {
234  auto admittance = config("admittance");
235  admittance("cop", copAdmittance);
236  admittance("maxVel", copMaxVel);
237  admittance("dfz", dfzAdmittance);
238  admittance("dfz_damping", dfzDamping);
239  }
240  if(config.has("dcm_tracking"))
241  {
242  auto dcmTracking = config("dcm_tracking");
243  if(dcmTracking.has("gains"))
244  {
245  dcmTracking("gains")("prop", dcmPropGain);
246  dcmTracking("gains")("integral", dcmIntegralGain);
247  dcmTracking("gains")("deriv", dcmDerivGain);
248  }
249  dcmTracking("derivator_time_constant", dcmDerivatorTimeConstant);
250  dcmTracking("integrator_time_constant", dcmIntegratorTimeConstant);
251  }
252  if(config.has("tasks"))
253  {
254  auto tasks = config("tasks");
255  if(tasks.has("com"))
256  {
257  tasks("com")("active_joints", comActiveJoints);
258  tasks("com")("stiffness", comStiffness);
259  tasks("com")("weight", comWeight);
260  tasks("com")("height", comHeight);
261  }
262 
263  if(tasks.has("torso"))
264  {
265 
266  tasks("torso")("pitch", torsoPitch);
267  tasks("torso")("stiffness", torsoStiffness);
268  tasks("torso")("weight", torsoWeight);
269  }
270 
271  if(tasks.has("pelvis"))
272  {
273  tasks("pelvis")("stiffness", pelvisStiffness);
274  tasks("pelvis")("weight", pelvisWeight);
275  }
276 
277  if(tasks.has("contact"))
278  {
279  if(tasks("contact").has("damping"))
280  {
281  double d = tasks("contact")("damping");
282  contactDamping = sva::MotionVecd({d, d, d}, {d, d, d});
283  }
284  if(tasks("contact").has("stiffness"))
285  {
286  double k = tasks("contact")("stiffness");
287  contactStiffness = sva::MotionVecd({k, k, k}, {k, k, k});
288  }
289  tasks("contact")("stiffness", contactStiffness);
290  tasks("contact")("weight", contactWeight);
291  }
292  }
293  if(config.has("vdc"))
294  {
295  auto vdc = config("vdc");
296  vdc("frequency", vdcFrequency);
297  vdc("stiffness", vdcStiffness);
298  }
299 
300  config("zmpcc", zmpcc);
301  }
